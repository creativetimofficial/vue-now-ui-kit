# Switch

To use the switch component, first import it: 
import {Switch} from 'src/components'
1

Global usage

Vue.component(Switch.name, Switch)
1

For local usage

export default {
  components: {
    [Switch.name]: Switch
  }
}

# Simple switch

<template>
  <div>
    <n-switch v-model="switches.defaultOn"></n-switch>
    <n-switch v-model="switches.defaultOff"></n-switch>
  </div>
</template>

<script>
  export default {
    data () {
        return {
           switches: {
             defaultOn: true,
             defaultOff: false
           }
        }
      }
  }
</script>

<script>

# Slots

<template>
  <div>
  <n-switch v-model="switches.defaultOn">
    <i class="fa fa-check" slot="on"></i>
    <i class="fab fa-exclamation" style="color:red;" slot="off"></i>
  </n-switch>
  </div>
</template>

<script>
  export default {
    data () {
        return {
           switches: {
             defaultOn: true
           }
        }
      }
  }
</script>
